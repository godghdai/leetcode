Merge two sorted linked lists and return it as a new list. The new list should be made by splicing together the nodes of the first two lists.

**Example:**
```
Input: 1->2->4, 1->3->4
Output: 1->1->2->3->4->4
```

**code**

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var mergeTwoLists = function(l1, l2) {
    var result=new ListNode(null);
    var node=null;
    var curr=result;
    while(l1||l2){
        if(l1==null){
            curr=curr.next=l2;           
            l2=l2.next
        }
        else if(l2==null){
            curr=curr.next=l1;
            l1=l1.next
        } 
        else if(l1.val<l2.val){
            curr=curr.next=l1;
            l1=l1.next
        }else
         {

            l2=l2.next
         }
        
    }
    return result.next
};
```
