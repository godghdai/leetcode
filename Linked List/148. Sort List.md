Sort a linked list in **O(n log n)** time using constant space complexity.

**Example 1:**
```
Input: 4->2->1->3
Output: 1->2->3->4
```
**Example 2:**
```
Input: -1->5->3->4->0
Output: -1->0->3->4->5
```

**code:**

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var sortList = function(head) {
    if(!head) return head;
    return sort(head);
};

var sort=function(list){
    var mid=getMidListNode(list);   
    if(mid==list&&!mid.next) return list;
    var l2=mid.next;
    mid.next=null;  
    return mergeTwoLists(sort(list),sort(l2));
}

var getMidListNode=function(list){
    if(!list.next||!list.next.next) return list;  
    var p1=list,p2=list.next;
    while(p2&&p2.next){
        p1=p1.next;
        p2=p2.next.next;
    }
    return p1;
}

var mergeTwoLists = function(l1, l2) {
    var result=new ListNode(null);
    var node=null;
    var curr=result;
    while(l1||l2){
        if(l1==null){
            curr=curr.next=l2;           
            l2=l2.next
        }
        else if(l2==null){
            curr=curr.next=l1;
            l1=l1.next
        } 
        else if(l1.val<l2.val){
            curr=curr.next=l1;
            l1=l1.next
        }else
         {
            curr=curr.next=l2;
            l2=l2.next
         }
        
    }
    return result.next
};
```
