Given a list of **unique** words, find all pairs of **distinct** indices (i, j) in the given list, so that the concatenation of the two words, i.e. words[i] + words[j] is a palindrome.

**Example 1**:
```
Input: ["abcd","dcba","lls","s","sssll"]
Output: [[0,1],[1,0],[3,2],[2,4]] 
Explanation: The palindromes are ["dcbaabcd","abcddcba","slls","llssssll"]
```

**Example 2**:
```
Input: ["bat","tab","cat"]
Output: [[0,1],[1,0]] 
Explanation: The palindromes are ["battab","tabbat"]
```

**code:**:

```js

/**
 * @param {string[]} words
 * @return {number[][]}
 */
var palindromePairs = function(words) {
    var map=new Map(),index,word;
    var len=words.length,ans=[];
    for(var i=0;i<len;i++)
        map.set(words[i],i);
    
    index=map.get("");
    if(index!=undefined){
        for(var i=0;i<len;i++)
            if(i!=index&&isPalindrome(words[i]))
                ans.push([i,index]);
    }
    
    var str1,str2;
    for(var i=0;i<len;i++){
        word=words[i]; 
        for(var wi=1;wi<=word.length;wi++){
            //[aba]ab=>ba[aba]ab
            str1=word.substring(0,wi);
            str2=word.substring(wi);''
            if(isPalindrome(str1)){
               index=map.get(reverse(str2));
               if(index!=undefined&&index!=i) 
                   ans.push([index,i]);
            }
            //ab[aba]=>ab[aba]ba;
            if(isPalindrome(str2)){
               index=map.get(reverse(str1));
               if(index!=undefined&&index!=i) 
                    ans.push([i,index]);
            }
        }
        
    }
    return ans;
};

var isPalindrome=function(word){
    var l=0,r=word.length-1;
    while(l<r){
        if(word[l]!=word[r])
            return false;
        l++;
        r--;
    }
    return true;
}

var reverse=function(word){
    var len=word.length-1;
    var res=new Array(word.length);
    for(var i=0;i<=len;i++)
        res[i]=word[len-i];
    return res.join("");
}
```
